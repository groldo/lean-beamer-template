% settings for title page =====================================================

\setbeamerfont{title}{size=\Large, series=\bfseries}
\setbeamercolor{titlegraphic}{bg=\seccol, fg=\fontcol}
\setbeamerfont{titlegraphic}{size=\Large, series=\bfseries}
\setbeamercolor{subtitle}{bg=\accseccol, fg=\maincol}
\setbeamerfont{subtitle}{size=\scriptsize, series=\bfseries}

\setbeamerfont{author}{size=\small, series=\bfseries}
\setbeamercolor{institute}{bg=\seccol, fg=\fontcol}
\setbeamerfont{institute}{size=\small}

% description of the title page

                \usebeamercolor[fg]{titlegraphic}
                \inserttitlegraphic

% settings for toc =============================================================
\setbeamercolor{section in toc}{bg=\seccol, fg=\fontcol}
\setbeamertemplate{section in toc}{
    \begin{beamercolorbox}{section in toc}
        \fcolorbox{\maincol}{\maincol}{
            \parbox[c][0.2cm][c]{0.2cm}{\color{\seccol}\bfseries\inserttocsectionnumber}
        }
        \fcolorbox{\seccol}{\seccol}{
            \parbox[c][0.2cm][c]{7.1cm}{\inserttocsection}
        }
    \end{beamercolorbox}
}

\setbeamercolor{subsection in toc}{bg=\seccol, fg=\fontcol}
\setbeamertemplate{subsection in toc}{
    \begin{beamercolorbox}{section in toc}
        %\hskip0.8cm
        \fcolorbox{\maincol}{\maincol}{
            \parbox[c][0.2cm][c]{0.5cm}{\color{\seccol}\inserttocsectionnumber.\inserttocsubsectionnumber}
        }
        \fcolorbox{\seccol}{\seccol}{
            \parbox[c][0.2cm][c]{6cm}{\inserttocsubsection}
        }
    \end{beamercolorbox}
}
% setting for toc =============================================================

% headline setting ============================================================
\setbeamerfont{section in head/foot}{size=\normalsize, series=\bfseries}
\setbeamercolor{section in head/foot}{bg=\seccol}
\setbeamerfont{subsection in head/foot}{size=\normalsize, series=\bfseries}
\setbeamercolor{subsection in head/foot}{bg=\seccol, fg=\maincol}

\setbeamertemplate{headline}{%
    \begin{beamercolorbox}[wd=\paperwidth,ht=0.3cm,center]{}
    \end{beamercolorbox}
    \begin{beamercolorbox}[wd=\paperwidth,right]{section in head/foot}
        \begin{minipage}[l][1cm][c]{0.75\textwidth}
            \makebox[0.8\textwidth][l]{
                \color{\maincol}\usebeamerfont{subsection in head/foot}\insertsubsection\par
            }
        \end{minipage}
        % Logo
        \begin{minipage}[c][1cm][c]{0.2\textwidth}
            \textcolor{\seccol}{titlegraphic}
            %\pgfuseimage{logo}
        \end{minipage}
        % Logo 
    \end{beamercolorbox}
}
% headline setting ============================================================

% footline setting ============================================================

\setbeamerfont{footline}{size=\tiny}

\setbeamertemplate{footline}{%
    \begin{beamercolorbox}[wd=\paperwidth,left]{}
        \progressbar@progressbar[\paperwidth][1pt](\progressbarcol,\footlinefontcol)
    \end{beamercolorbox}
    \begin{beamercolorbox}[wd=\paperwidth,ht=1ex,center]{}
    \end{beamercolorbox}
    \begin{beamercolorbox}[wd=\paperwidth,ht=3ex,dp=0.5ex,center]{}
    \end{beamercolorbox}
    \begin{beamercolorbox}[sep=2ex,wd=\paperwidth,ht=1ex,center]{}
        \begin{beamercolorbox}[wd=0.2\paperwidth,dp=1pt,left]{}
            \insertdate
        \end{beamercolorbox}
        %\hskip1cm
        \begin{beamercolorbox}[wd=0.4\paperwidth,dp=1pt,center]{}
            \insertsection
        \end{beamercolorbox}
        %\hskip1cm
        \begin{beamercolorbox}[wd=0.2\paperwidth,dp=1pt,right]{}
            \insertframenumber
        \end{beamercolorbox}
    \end{beamercolorbox}
    }%
